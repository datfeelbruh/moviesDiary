# Installation
1. git clone --recurse-submodules https://github.com/datfeelbruh/moviesDiary.git
2. cd ./moviesDiary
3. git checkout tls-init
4. docker-compose up -d --build
5. git checkout tls-renew
6. docker-compose up frontend -d --build
